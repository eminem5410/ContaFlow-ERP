Modelo de Datos — ERP Contable Multi-Tenant
PostgreSQL Schema con aislamiento por tenant_id
erDiagram
    TENANTS {
        uuid id PK
        string name
        string cuit
        string plan
        boolean active
        timestamp created_at
    }

    USERS {
        uuid id PK
        uuid tenant_id FK
        string email
        string password_hash
        string role
        boolean active
        timestamp created_at
    }

    ACCOUNTS {
        uuid id PK
        uuid tenant_id FK
        string code
        string name
        string type
        string subtype
        boolean imputable
        boolean active
        timestamp created_at
    }

    JOURNAL_ENTRIES {
        uuid id PK
        uuid tenant_id FK
        string number
        date date
        string description
        string status
        uuid created_by FK
        timestamp created_at
    }

    JOURNAL_LINES {
        uuid id PK
        uuid entry_id FK
        uuid account_id FK
        decimal debit
        decimal credit
        string description
    }

    INVOICES {
        uuid id PK
        uuid tenant_id FK
        string type
        string number
        uuid customer_id FK
        date date
        date due_date
        decimal total
        string status
        string afip_result
        timestamp created_at
    }

    INVOICE_ITEMS {
        uuid id PK
        uuid invoice_id FK
        string description
        decimal quantity
        decimal unit_price
        decimal subtotal
        decimal vat_rate
    }

    CUSTOMERS {
        uuid id PK
        uuid tenant_id FK
        string name
        string cuit_dni
        string email
        string phone
        string address
        timestamp created_at
    }

    SUPPLIERS {
        uuid id PK
        uuid tenant_id FK
        string name
        string cuit
        string email
        string phone
        string address
        timestamp created_at
    }

    PAYMENTS {
        uuid id PK
        uuid tenant_id FK
        uuid customer_id FK
        decimal amount
        date date
        string method
        string reference
        string status
        timestamp created_at
    }

    BANK_ACCOUNTS {
        uuid id PK
        uuid tenant_id FK
        string name
        string bank_name
        string account_number
        string currency
        decimal balance
    }

    BANK_MOVEMENTS {
        uuid id PK
        uuid bank_account_id FK
        uuid tenant_id FK
        date date
        decimal amount
        string type
        string description
        boolean reconciled
    }

    TAX_CONFIGURATIONS {
        uuid id PK
        uuid tenant_id FK
        string tax_type
        decimal rate
        string category
        boolean active
    }

    AUDIT_LOGS {
        uuid id PK
        uuid tenant_id FK
        uuid user_id FK
        string entity_type
        uuid entity_id
        string action
        jsonb old_values
        jsonb new_values
        timestamp created_at
    }
